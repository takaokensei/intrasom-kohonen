\begin{frame}{1. O Desafio da Alta Dimensionalidade}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item \textbf{Problema:} Como visualizar e agrupar dados complexos sem depender de rótulos?
        \item \textbf{Limitações Comuns:}
          \begin{itemize}
            \item \textbf{PCA:} Linear, ignora correlações não-lineares.
            \item \textbf{t-SNE / UMAP:} Ótima separação visual, mas perde vizinhança contínua e não cria protótipos paramétricos.
          \end{itemize}
        \item \textbf{A Solução:} SOM (Self-Organizing Maps).
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      \begin{beamercolorbox}[wd=\linewidth,rounded=true,center,sep=2pt]{frametitle}
        \includegraphics[width=0.96\linewidth]{outputs/figures/dim_reduction_comparison.png}
      \end{beamercolorbox}
      
      \vspace{0.1cm}
      \scriptsize \textcolor{tokyoBlue}{Fig. 1: PCA vs t-SNE vs UMAP}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{2. SOM em 60 Segundos: Intuição Visual}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textbf{Grade Organizada:} Neurônios dispostos em malha hexagonal 2D.
        \item \textbf{Neurônios-Protótipo:} Cada hexágono representa um grupo de dados semelhantes.
        \item \textbf{Vizinhos Próximos:} A proximidade física na grade reflete a similaridade real no espaço original.
        \item \textbf{Topologia Hexagonal:} 6 vizinhos equidistantes na grade, otimizando o fluxo de atualização.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \centering
      \begin{beamercolorbox}[wd=\linewidth,rounded=true,center,sep=2pt]{frametitle}

\end{frame}

\begin{frame}{3. Os Datasets de Teste}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item \textbf{Synthetic Control (Séries):}
          \begin{itemize}
            \item 600 séries temporais, 60 atributos.
            \item 6 classes de tendência (Normal, Cíclica, Crescente, Decrescente, etc.).
            \item Malhas testadas: 5x5 a 20x20.
          \end{itemize}
        \item \textbf{20 Newsgroups (Textos):}
          \begin{itemize}
            \item 400 notícias, 4 classes temáticas (Graphics, Space, Baseball, Mideast).
            \item Grade base hexagonal 10x10.
          \end{itemize}
      \end{itemize}
    \end{column}

  \begin{columns}

\begin{frame}{7. Resultado 3: Representação de Texto}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item \textbf{TF-IDF (ARI: 0.1695, NMI: 0.2844):} Abordagem Bag-of-Words. Falha ao lidar com sinônimos e contexto semântico.
        \item \textbf{Sentence-BERT (ARI: 0.5873, NMI: 0.6027):} Embeddings contextuais densos.
        \item \textbf{Vantagem do SBERT:} Agrupa ideias e significado conceitual de forma extremamente coesa na grade de Kohonen.
      \end{itemize}

\begin{frame}{9. Conclusão}
  \begin{itemize}
    \item \textbf{Conclusão:} O SOM de Kohonen se consolida como uma excelente alternativa para redução de dimensionalidade com alta interpretabilidade.
    \item \textbf{Destaques:}
      \begin{itemize}
        \item Erro topográfico mínimo, garantindo organização geométrica correta.
        \item Sucesso absoluto das representações semânticas profundas (SBERT) para textos.
        \item Dashboard web robusto integrado, 100\% serverless e funcional.
      \end{itemize}
    \item \textbf{Trabalhos Futuros:} Implementação de grades toroidais tridimensionais.
  \end{itemize}
